127 BOOL LoadTGA(TextureImage *texture, char *filename) // Loads A TGA File Into Memory
128 {
129  GLubyte TGAheader[12]={0,0,2,0,0,0,0,0,0,0,0,0}; // Uncompressed TGA Header
130  GLubyte TGAcompare[12]; // Used To Compare TGA Header
131  GLubyte header[6]; // First 6 Useful Bytes From The Header
132  GLuint bytesPerPixel; // Holds Number Of Bytes Per Pixel Used In The TGA File
133  GLuint imageSize; // Used To Store The Image Size When Setting Aside Ram
134  GLuint temp; // Temporary Variable
135  GLuint type=GL_RGBA; // Set The Default GL Mode To RBGA (32 BPP)
136  int i=0;
137 
138  FILE *file = fopen(filename, "rb"); // Open The TGA File
139 
140  if( file==NULL || // Does File Even Exist?
141  fread(TGAcompare,1,sizeof(TGAcompare),file)!=sizeof(TGAcompare) || // Are There 12 Bytes To Read?
142  memcmp(TGAheader,TGAcompare,sizeof(TGAheader))!=0 || // Does The Header Match What We Want?
143  fread(header,1,sizeof(header),file)!=sizeof(header)) // If So Read Next 6 Header Bytes
144  {
145  if (file == NULL) // Did The File Even Exist? *Added Jim Strong*
146  return FALSE; // Return FALSE
147  else
148  {
149  fclose(file); // If Anything Failed, Close The File
150  return FALSE; // Return FALSE
151  }
152  }
153 
154  texture->width = header[1] * 256 + header[0]; // Determine The TGA Width (highbyte*256+lowbyte)
155  texture->height = header[3] * 256 + header[2]; // Determine The TGA Height (highbyte*256+lowbyte)
156 
157  if( texture->width <=0 || // Is The Width Less Than Or Equal To Zero
158  texture->height <=0 || // Is The Height Less Than Or Equal To Zero
159  (header[4]!=24 && header[4]!=32)) // Is The TGA 24 or 32 Bit?
160  {
161  fclose(file); // If Anything Failed, Close The File
162  return FALSE; // Return FALSE
163  }
164 
165  texture->bpp = header[4]; // Grab The TGA's Bits Per Pixel (24 or 32)
166  bytesPerPixel = texture->bpp/8; // Divide By 8 To Get The Bytes Per Pixel
167  imageSize = texture->width*texture->height*bytesPerPixel; // Calculate The Memory Required For The TGA Data
168 
169  texture->imageData=(GLubyte *)malloc(imageSize); // Reserve Memory To Hold The TGA Data
170 
171  if( texture->imageData==NULL || // Does The Storage Memory Exist?
172  fread(texture->imageData, 1, imageSize, file)!=imageSize) // Does The Image Size Match The Memory Reserved?
173  {
174  if(texture->imageData!=NULL) // Was Image Data Loaded
175  free(texture->imageData); // If So, Release The Image Data
176 
177  fclose(file); // Close The File
178  return FALSE; // Return FALSE
179  }
180 
181  for(i=0; i<(int)(imageSize); i+=bytesPerPixel) // Loop Through The Image Data
182  { // Swaps The 1st And 3rd Bytes ('R'ed and 'B'lue)
183  temp=texture->imageData[i]; // Temporarily Store The Value At Image Data 'i'
184  texture->imageData[i] = texture->imageData[i + 2]; // Set The 1st Byte To The Value Of The 3rd Byte
185  texture->imageData[i + 2] = temp; // Set The 3rd Byte To The Value In 'temp' (1st Byte Value)
186  }
187 
188  fclose (file); // Close The File
189 
190  // Build A Texture From The Data
191  glGenTextures(1, &texture[0].texID); // Generate OpenGL texture IDs
192 
193  glBindTexture(GL_TEXTURE_2D, texture[0].texID); // Bind Our Texture
194  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR); // Linear Filtered
195  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR); // Linear Filtered
196 
197  if (texture[0].bpp==24) // Was The TGA 24 Bits
198  {
199  type=GL_RGB; // If So Set The 'type' To GL_RGB
200  }
201 
202  glTexImage2D(GL_TEXTURE_2D, 0, type, texture[0].width, texture[0].height, 0, type, GL_UNSIGNED_BYTE, texture[0].imageData);
203 
204  return TRUE; // Texture Building Went Ok, Return True
205 }

207 GLvoid BuildFont(GLvoid) // Build Our Font Display List
208 {
209  int loop1=0;
210 
211  base=glGenLists(256); // Creating 256 Display Lists
212  glBindTexture(GL_TEXTURE_2D, textures[0].texID); // Select Our Font Texture
213  for (loop1=0; loop1<256; loop1++) // Loop Through All 256 Lists
214  {
215  float cx=(float)(loop1%16)/16.0f; // X Position Of Current Character
216  float cy=(float)(loop1/16)/16.0f; // Y Position Of Current Character
217 
218  glNewList(base+loop1,GL_COMPILE); // Start Building A List
219  glBegin(GL_QUADS); // Use A Quad For Each Character
220  glTexCoord2f(cx,1.0f-cy-0.0625f); // Texture Coord (Bottom Left)
221  glVertex2d(0,16); // Vertex Coord (Bottom Left)
222  glTexCoord2f(cx+0.0625f,1.0f-cy-0.0625f); // Texture Coord (Bottom Right)
223  glVertex2i(16,16); // Vertex Coord (Bottom Right)
224  glTexCoord2f(cx+0.0625f,1.0f-cy-0.001f); // Texture Coord (Top Right)
225  glVertex2i(16,0); // Vertex Coord (Top Right)
226  glTexCoord2f(cx,1.0f-cy-0.001f); // Texture Coord (Top Left)
227  glVertex2i(0,0); // Vertex Coord (Top Left)
228  glEnd(); // Done Building Our Quad (Character)
229  glTranslated(14,0,0); // Move To The Right Of The Character
230  glEndList(); // Done Building The Display List
231  } // Loop Until All 256 Are Built
232 }
